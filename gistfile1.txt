#Daily Low Line
plot LowWeekly = low(period = AggregationPeriod.WEEK);
LowWeekly.SetDefaultColor(Color.RED);
LowWeekly.SetLineWeight(1);

#Daily High Line
plot HighWeekly = high(period = AggregationPeriod.WEEK);
HighWeekly.SetDefaultColor(Color.GREEN);
HighWeekly.SetLineWeight(1);
